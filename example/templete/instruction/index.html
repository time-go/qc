Module Development
  • qc use "need" to inlude a file
  • relative path:need("$url:xxxxx"),need("$url:../xxxxx")
  • absolute path:need("$url:/xxxxx")
  • include text:need("$url:xxxxx!text")
  • out means export

    out=function () {
        console.log("hello")
    }

    var Test = need("$url:test.js");
